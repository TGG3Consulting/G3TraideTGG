pythonclass SignalOptimizer:
    async def optimize(self, signal: TradeSignal) -> OptimizedSignal | None:
        """
        Оптимизировать сигнал.
        
        Returns:
            OptimizedSignal если оптимизация успешна, None если сигнал отклонён
        """
        try:
            # Получить рыночные данные
            market_data = await self._get_market_data(signal.symbol)
            if market_data is None:
                self.logger.warning("no_market_data", symbol=signal.symbol)
                return None
            
            # Извлечь фичи
            features = self.feature_engineer.extract_features(
                signal.symbol, 
                signal.timestamp
            )
            
            # Получить предсказания
            predictions = self.model.predict(features)
            
            # Проверить уверенность
            if predictions.confidence < self.config.min_confidence:
                self.logger.info(
                    "signal_rejected_low_confidence",
                    symbol=signal.symbol,
                    confidence=predictions.confidence,
                    threshold=self.config.min_confidence
                )
                return None
            
            # Создать оптимизированный сигнал
            optimized = self._build_optimized_signal(signal, predictions, market_data)
            
            self.logger.info(
                "signal_optimized",
                symbol=signal.symbol,
                original_sl=signal.stop_loss,
                optimized_sl=optimized.stop_loss,
                confidence=predictions.confidence
            )
            
            return optimized
            
        except DataNotFoundError as e:
            self.logger.warning("optimization_failed_no_data", error=str(e))
            return None
        except Exception as e:
            self.logger.error("optimization_failed", error=str(e), exc_info=True)
            raise  # Пробрасываем неожиданные ошибки!

### ОБЯЗАТЕЛЬНЫЕ ПРОВЕРКИ ML:

1. РАЗДЕЛЕНИЕ ДАННЫХ СТРОГО ПО ВРЕМЕНИ:
   ❌ random split (train_test_split с shuffle=True)
   ✅ temporal split (первые 80% по времени = train, последние 20% = test)
```python
# ПРАВИЛЬНО:
data = data.sort_values('timestamp')
split_idx = int(len(data) * 0.8)
train = data.iloc[:split_idx]
test = data.iloc[split_idx:]  # Только БУДУЩИЕ данные!

# НЕПРАВИЛЬНО:
```

2. WALK-FORWARD VALIDATION:
   - Обучить на месяц 1-3
   - Тестировать на месяц 4
   - Обучить на месяц 1-4
   - Тестировать на месяц 5
   - И так далее

3. МЕТРИКИ НА OUT-OF-SAMPLE:
   - Все метрики считать ТОЛЬКО на test данных
   - Train метрики НЕ показывать пользователю
   - Если test metrics сильно хуже train — OVERFITTING!

4. МИНИМАЛЬНЫЕ ТРЕБОВАНИЯ:
   - Минимум 500 сигналов для обучения
   - Минимум 100 сигналов для теста
   - Минимум 3 месяца исторических данных
   - Если меньше — НЕ обучать, вернуть ошибку!
```python
def validate_data_sufficiency(train_data: pd.DataFrame, test_data: pd.DataFrame) -> None:
    """Проверить достаточность данных."""
    if len(train_data) < 500:
        raise InsufficientDataError(f"Need 500+ train samples, got {len(train_data)}")
    if len(test_data) < 100:
        raise InsufficientDataError(f"Need 100+ test samples, got {len(test_data)}")
    
    train_period = (train_data['timestamp'].max() - train_data['timestamp'].min()).days
    if train_period < 90:
        raise InsufficientDataError(f"Need 90+ days of data, got {train_period}")
```

5. BASELINE СРАВНЕНИЕ:
   - Всегда сравнивать с простым baseline
   - Baseline: случайный выбор или всегда LONG
   - Если ML не лучше baseline на 10%+ — модель бесполезна!
```python
def calculate_baseline(test_data: pd.DataFrame) -> float:
    """Baseline: просто держать все сигналы без оптимизации."""
    return test_data['pnl_percent'].mean()

def validate_model_vs_baseline(model_pnl: float, baseline_pnl: float) -> bool:
    """Модель должна быть лучше baseline минимум на 10%."""
    improvement = (model_pnl - baseline_pnl) / abs(baseline_pnl) * 100
    if improvement < 10:
        logger.warning("MODEL_NOT_BETTER_THAN_BASELINE", 
                      model=model_pnl, baseline=baseline_pnl, improvement=improvement)
        return False
    return True
```

6. РЕГУЛЯРИЗАЦИЯ:

   - Early stopping

```python
class MarketImpactModel:

@dataclass
class FullTransactionCosts:
    """Полные транзакционные издержки."""
    
    # Комиссии
    maker_fee: Decimal = Decimal("0.0002")
    taker_fee: Decimal = Decimal("0.0005")
    
    # Funding (средний за период)
    avg_funding_rate: Decimal = Decimal("0.0001")  # 0.01% каждые 8ч
    
    # Spread
    avg_spread_pct: float = 0.0005  # 0.05%
    
    def calculate_total_cost(
        self,
        entry_price: Decimal,
        exit_price: Decimal,
        hold_time_hours: float,
        is_maker_entry: bool,
        is_maker_exit: bool
    ) -> Decimal:
        """Рассчитать ВСЕ издержки."""
        
        # Entry cost
        entry_fee = self.maker_fee if is_maker_entry else self.taker_fee
        entry_spread = Decimal(str(self.avg_spread_pct / 2))
        
        # Exit cost  
        exit_fee = self.maker_fee if is_maker_exit else self.taker_fee
        exit_spread = Decimal(str(self.avg_spread_pct / 2))
        
        # Funding (сколько 8-часовых периодов)
        funding_periods = int(hold_time_hours / 8)
        funding_cost = self.avg_funding_rate * funding_periods
        
        total = entry_fee + exit_fee + entry_spread + exit_spread + funding_cost
        return total

        base_size: Decimal,
        consecutive_losses: int
    ) -> Decimal:

            reduction = 0.5 ** (consecutive_losses - 2)  # 50%, 25%, 12.5%...
            return base_size * Decimal(str(reduction))
        return base_size

        self, 
        features: FeatureVector,
        prediction: Prediction
    ) -> Explanation:
        """
        Вернуть:
        1. Top-5 фичей повлиявших на решение
        2. SHAP values для каждой фичи
        3. Человекочитаемое объяснение
        """

        import shap
        explainer = shap.TreeExplainer(self.model)
        shap_values = explainer.shap_values(features)
        
        # Top features
        top_features = sorted(
            zip(self.feature_names, shap_values),
            key=lambda x: abs(x[1]),
            reverse=True
        )[:5]
        
        # Человекочитаемое объяснение
        explanation = self._generate_explanation(top_features)
        
        return Explanation(
            top_features=top_features,
            shap_values=shap_values,
            human_readable=explanation
        )
```
